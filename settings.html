// === Global Sales Agent : 이메일 발송 + Google Sheets 동기화 통합본 ===

// 데이터 저장용 시트 (없으면 자동 생성)
function getSheet() {
  var props = PropertiesService.getScriptProperties();
  var id = props.getProperty('SHEET_ID');
  var ss;
  if (id) { ss = SpreadsheetApp.openById(id); }
  else {
    ss = SpreadsheetApp.create('GlobalSalesData');
    props.setProperty('SHEET_ID', ss.getId());
  }
  var sh = ss.getSheetByName('AppData');
  if (!sh) { sh = ss.insertSheet('AppData'); }
  return sh;
}

function readAll() {
  var rows = getSheet().getDataRange().getValues();
  var out = {};
  rows.forEach(function(r) {
    if (r[0]) { try { out[r[0]] = JSON.parse(r[1] || '[]'); } catch(e) { out[r[0]] = []; } }
  });
  return out;
}

function writeKey(key, value) {
  var sh = getSheet();
  var rows = sh.getDataRange().getValues();
  for (var i = 0; i < rows.length; i++) {
    if (rows[i][0] === key) { sh.getRange(i + 1, 2).setValue(JSON.stringify(value)); return; }
  }
  sh.appendRow([key, JSON.stringify(value)]);
}

function json(o) {
  return ContentService.createTextOutput(JSON.stringify(o))
    .setMimeType(ContentService.MimeType.JSON);
}

// 데이터 불러오기 (모든 기기 공통)
function doGet(e) {
  return json(readAll());
}

// 데이터 저장 또는 이메일 발송
function doPost(e) {
  try {
    var d = JSON.parse(e.postData.contents);
    if (d.action === 'sync') {
      writeKey(d.key, d.value);
      return json({ status: 'ok' });
    }
    // 그 외에는 이메일 발송
    GmailApp.sendEmail(d.to, d.subject, d.body, {
      name: d.senderName || Session.getActiveUser().getEmail()
    });
    return json({ status: 'ok' });
  } catch(err) {
    return json({ status: 'error', message: err.message });
  }
}
